configureStore.js 
1    import { createStore, applyMiddleware } from 'redux' 
2    import rootReducer from '../reducers' 
3    import { ping } from './enhancers/ping' // <!-- подключаем наш enhancer 
4    import createLogger from 'redux-logger' 
5     
6    export default function configureStore(initialState) { 
7      const logger = createLogger(); 
8      const store = createStore( 
9          rootReducer, 
10         initialState, 
11         applyMiddleware(ping), // <!-- добавляем enhancer в цепочку middleware'ов 
12         applyMiddleware(logger)); 
13    
14     //A store holds the whole state tree of your application. 
15     //The only way to change the state inside it is to dispatch an action on it. 
16     //A store is not a class. It’s just an object with a few methods on it. 
17     //To create it, pass your root reducing function to createStore. 
18     //http://redux.js.org/docs/api/Store.html 
19    
20     //Store Methods 
21    
22   //getState() 
23   //dispatch(action) 
24   //subscribe(listener) 
25   //replaceReducer(nextReducer)  Не используется в обычной среде 
26    
27     if (module.hot) { 
28       module.hot.accept('../reducers', () => {  //module.hot.accept('../reducers', function () { 
29         const nextRootReducer = require('../reducers'); 
30         store.replaceReducer(nextRootReducer); 
31       }); 
32     } 
33    
34     return store; 
35   }
