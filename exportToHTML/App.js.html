App.js 
1    import React, { Component } from 'react' 
2    import { bindActionCreators } from 'redux' 
3    import { connect } from 'react-redux' 
4    import User from '../components/User' 
5    import Page from '../components/Page' 
6    import * as pageActions from '../actions/PageActions' 
7     
8    class App extends Component { 
9      render() { 
10       const { user, page } = this.props; 
11       const { setYear } = this.props.pageActions; 
12    
13       return <div> 
14         <User name={user.name} /> 
15         <Page photos={page.photos} year={page.year} setYear={setYear} /> 
16       </div> 
17     } 
18   } 
19    
20   //Устанавливаем соответствие глобального state props каждого компонента 
21   function mapStateToProps(state) { 
22     return { 
23       user: state.user, 
24       page: state.page 
25     } 
26   } 
27    
28    
29    
30   function mapDispatchToProps(dispatch) { 
31     return { 
32       pageActions: bindActionCreators(pageActions, dispatch) 
33       //we bind Action Creator to dispatch http://redux.js.org/docs/Glossary.html#action-creator 
34       //this created action is immediately dispatched 
35     } 
36   } 
37    
38    
39    
40   export default connect(mapStateToProps, mapDispatchToProps)(App)
